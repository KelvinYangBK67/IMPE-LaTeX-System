\cs_new:Npn \__font_hindi_select:n #1
  {
    \edef\tempfamily{\HIfamily}%
    \edef\tempseries{\HIseries}%
    \ifx\tempfamily\rmname
      \ifx\tempseries\bfname \HIRmBf \else \HIRmMd \fi
    \else
      \ifx\tempseries\bfname \HISfBf \else \HISfMd \fi
    \fi
  }

    \fi
  }

\cs_new_protected:Npn \__font_special_rtl_wrap:nn #1#2
  {
    \ifhmode
      #1{#2}%
    \else
      \par
      \begingroup
      \leftskip=0pt plus 1fil\relax
      \rightskip=0pt\relax
      \parfillskip=0pt\relax
      \noindent
      \hbox to \linewidth { \hfill \beginR #1{#2}\endR }
      \par
      \endgroup
    \fi
  }

\cs_new_protected:Npn \__font_special_rtl_wrap_plain:n #1
  {
    \ifhmode
      \RL{#1}%
    \else
      \par
      \begingroup
      \leftskip=0pt plus 1fil\relax
      \rightskip=0pt\relax
      \parfillskip=0pt\relax
      \noindent
      \hbox to \linewidth { \hfill \beginR #1\endR }
      \par
      \endgroup
    \fi
  }

\cs_new_protected:Npn \__font_define_special_psalter_pahlavi:
  {
    \newfontfamily\PSPRmMd[
      Path = \l__font_path_tl,
      RawFeature = { script=phlp }
    ] { \l__font_regular_tl }

    \DeclareDocumentCommand{\PSP}{m}
      {
        {
          \def\PSPtextit####1{{\PSPRmMd ####1}}
          \def\PSPtextbf####1{{\PSPRmMd ####1}}
          \def\PSPtextsf####1{{\PSPRmMd ####1}}
          \def\PSPtexttt####1{{\PSPRmMd ####1}}
          \let\textit\PSPtextit
          \let\textbf\PSPtextbf
          \let\textsf\PSPtextsf
          \let\texttt\PSPtexttt
          \PSPRmMd
          \__font_special_rtl_wrap_plain:n { ##1 }
        }
      }
  }

\cs_new_protected:Npn \__font_vertical_embed_word:nnn #1#2#3
  {
    \hbox_set:Nn \l__font_vertical_embed_box
      {
        \rotatebox[origin=c]{-90}
          {
            { #2 #3 }
          }
      }
    \raisebox
      {
        \dim_eval:n
          {
            \ht\strutbox
            - \box_ht:N \l__font_vertical_embed_box
            + \l__font_vertical_embed_top_correction_dim
            - (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 16
            + (#1)
          }
      }
      [
        \dim_eval:n
          {
            \box_ht:N \l__font_vertical_embed_box
            - \ht\strutbox
            - \l__font_vertical_embed_top_correction_dim
            + (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 16
            - (#1)
          }
      ]
      [
        \dim_eval:n
          {
            \box_dp:N \l__font_vertical_embed_box
            + \dp\strutbox
            + \l__font_vertical_embed_top_correction_dim
            + (#1)
            + \l__font_vertical_embed_bottom_padding_dim
            + (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 8
          }
      ]
      { \box_use:N \l__font_vertical_embed_box }
  }

\cs_new_protected:Npn \__font_vertical_embed_words:nnn #1#2#3
  {
    \leavevmode
    \seq_set_split:Nnn \l__font_mongolian_words_seq { ~ } {#3}
    \seq_map_inline:Nn \l__font_mongolian_words_seq
      {
        \__font_vertical_embed_word:nnn {#1} {#2} {##1}
        \hspace{\l__font_vertical_embed_gap_tl}
      }
  }

\makeatletter
\def\HIinitstate{%
  \edef\HIfamily{rm}%
  \edef\HIseries{md}%
  \edef\HI@fam{\f@family}%
  \edef\HI@sf{\sfdefault}%
  \ifx\HI@fam\HI@sf \edef\HIfamily{sf}\fi
  \edef\HI@series{\f@series}%
  \def\HI@b{b}%
  \def\HI@bx{bx}%
  \ifx\HI@series\HI@b \edef\HIseries{bf}\fi
  \ifx\HI@series\HI@bx \edef\HIseries{bf}\fi
}

\def\TIinitstate{%
  \edef\TIfamily{rm}%
  \edef\TIseries{md}%
  \edef\TI@shape{\f@shape}%
  \def\TI@it{it}%
  \ifx\TI@shape\TI@it \edef\TIfamily{it}\fi
  \edef\TI@series{\f@series}%
  \def\TI@b{b}%
  \def\TI@bx{bx}%
  \ifx\TI@series\TI@b \edef\TIseries{bf}\fi
  \ifx\TI@series\TI@bx \edef\TIseries{bf}\fi
}
\makeatother
