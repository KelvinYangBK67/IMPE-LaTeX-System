% Mongolian special builders.

\RequirePackage{graphicx}
\RequirePackage[export]{adjustbox}

\ExplSyntaxOn

\seq_new:N \l__font_mongolian_words_seq
\tl_new:N \l__font_vertical_embed_gap_tl
\dim_new:N \l__font_vertical_embed_top_correction_dim
\dim_new:N \l__font_vertical_embed_bottom_padding_dim
\box_new:N \l__font_vertical_embed_box

\cs_new_protected:Npn \__font_vertical_embed_word:nnn #1#2#3
  {
    \hbox_set:Nn \l__font_vertical_embed_box
      {
        \rotatebox[origin=c]{-90}
          {
            { #2 #3 }
          }
      }
    \raisebox
      {
        \dim_eval:n
          {
            \ht\strutbox
            - \box_ht:N \l__font_vertical_embed_box
            + \l__font_vertical_embed_top_correction_dim
            - (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 16
            + (#1)
          }
      }
      [
        \dim_eval:n
          {
            \box_ht:N \l__font_vertical_embed_box
            - \ht\strutbox
            - \l__font_vertical_embed_top_correction_dim
            + (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 16
            - (#1)
          }
      ]
      [
        \dim_eval:n
          {
            \box_dp:N \l__font_vertical_embed_box
            + \dp\strutbox
            + \l__font_vertical_embed_top_correction_dim
            + (#1)
            + \l__font_vertical_embed_bottom_padding_dim
            + (
                \box_ht:N \l__font_vertical_embed_box
                + \box_dp:N \l__font_vertical_embed_box
              ) / 8
          }
      ]
      { \box_use:N \l__font_vertical_embed_box }
  }

\cs_new_protected:Npn \__font_vertical_embed_words:nnn #1#2#3
  {
    \leavevmode
    \seq_set_split:Nnn \l__font_mongolian_words_seq { ~ } {#3}
    \seq_map_inline:Nn \l__font_mongolian_words_seq
      {
        \__font_vertical_embed_word:nnn {#1} {#2} {##1}
        \hspace{\l__font_vertical_embed_gap_tl}
      }
  }

\cs_new_protected:Npn \__font_define_special_mongolian:
  {
    \newfontfamily\MOfont[
      Path = \l__font_path_tl,
      RawFeature = { script=mong }
    ] { \l__font_regular_tl }

    \tl_set:Nn \l__font_vertical_embed_gap_tl { 0.35em }
    \dim_set:Nn \l__font_vertical_embed_top_correction_dim { -0.42em }
    \dim_set:Nn \l__font_vertical_embed_bottom_padding_dim { 0.30em }

    \DeclareDocumentCommand{\MO}{m}
      {
        {
          \def\MOtextsf####1{{\MOfont ####1}}
          \def\MOtextit####1{{\MOfont ####1}}
          \def\MOtextbf####1{{\MOfont ####1}}
          \let\textsf\MOtextsf
          \let\textit\MOtextit
          \let\textbf\MOtextbf
          \MOfont ##1
        }
      }

    \DeclareDocumentCommand{\MOV}{ O{0pt} m }
      {
        \__font_vertical_embed_words:nnn {##1} { \MOfont } {##2}
      }
  }

\ExplSyntaxOff
