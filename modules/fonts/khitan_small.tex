% Khitan Small Script special builder.
%
% The current TeX/fontspec shaping path does not reliably expose the font's
% native cluster substitution behaviour, so NexTeX composes Khitan Small Script
% clusters explicitly from the encoded character stream.
%
% Model used here:
% - clusters are separated by spaces
% - Type A starts with two adjacent characters
% - Type B inserts U+16FE4 KHITAN SMALL SCRIPT FILLER after the first character
% - Type A lays out rows of pairs, with a final centred character if odd
% - Type B lays out a first centred character, then applies the Type A rule to
%   the remaining characters

\ExplSyntaxOn

\tl_const:Nn \c__font_khs_filler_tl { 𖿤 }

\seq_new:N \l__font_khs_cluster_seq
\seq_new:N \l__font_khs_char_seq
\dim_new:N \l__font_khs_cell_wd_dim
\dim_new:N \l__font_khs_cell_ht_dim
\dim_new:N \l__font_khs_cluster_wd_dim
\dim_new:N \l__font_khs_row_gap_dim
\dim_new:N \l__font_khs_col_gap_dim
\dim_new:N \l__font_khs_space_gap_dim
\dim_new:N \l__font_khs_target_ht_dim
\dim_new:N \l__font_khs_actual_ht_dim
\dim_new:N \l__font_khs_raise_dim
\bool_new:N \l__font_khs_type_b_bool
\bool_new:N \l__font_khs_first_row_bool
\bool_new:N \l__font_khs_first_cluster_bool
\int_new:N \l__font_khs_run_max_rows_int
\int_new:N \l__font_khs_cluster_rows_int
\box_new:N \l__font_khs_measure_box
\box_new:N \l__font_khs_cluster_box
\cs_new_protected:Npn \__font_khs_measure_cells:
  {
    \dim_zero:N \l__font_khs_cell_wd_dim
    \dim_zero:N \l__font_khs_cell_ht_dim
    \seq_map_inline:Nn \l__font_khs_char_seq
      {
        \hbox_set:Nn \l__font_khs_measure_box { ##1 }
        \dim_set:Nn \l_tmpa_dim { \box_wd:N \l__font_khs_measure_box }
        \dim_set:Nn \l_tmpb_dim
          { \box_ht:N \l__font_khs_measure_box + \box_dp:N \l__font_khs_measure_box }
        \dim_compare:nNnT { \l_tmpa_dim } > { \l__font_khs_cell_wd_dim }
          { \dim_set_eq:NN \l__font_khs_cell_wd_dim \l_tmpa_dim }
        \dim_compare:nNnT { \l_tmpb_dim } > { \l__font_khs_cell_ht_dim }
          { \dim_set_eq:NN \l__font_khs_cell_ht_dim \l_tmpb_dim }
      }
    \dim_set:Nn \l__font_khs_col_gap_dim { 0.2em }
    \dim_set:Nn \l__font_khs_row_gap_dim { 0.05em }
    \dim_set:Nn \l__font_khs_space_gap_dim { 0.5em }
    \dim_set:Nn \l__font_khs_cluster_wd_dim
      { 2\l__font_khs_cell_wd_dim + \l__font_khs_col_gap_dim }
  }

\cs_new_protected:Npn \__font_khs_set_run_max_rows:n #1
  { \int_set:Nn \l__font_khs_run_max_rows_int {#1} }

\cs_new_protected:Npn \__font_khs_compute_cluster_rows:
  {
    \bool_if:NTF \l__font_khs_type_b_bool
      {
        \int_set:Nn \l__font_khs_cluster_rows_int
          { 1 + \int_div_truncate:nn { \seq_count:N \l__font_khs_char_seq } { 2 } }
      }
      {
        \int_set:Nn \l__font_khs_cluster_rows_int
          {
            \int_div_truncate:nn
              { \seq_count:N \l__font_khs_char_seq + 1 }
              { 2 }
          }
      }
  }

\cs_new_protected:Npn \__font_khs_center_row:n #1
  {
    \bool_if:NF \l__font_khs_first_row_bool
      { \kern \l__font_khs_row_gap_dim }
    \bool_set_false:N \l__font_khs_first_row_bool
    \hbox to \l__font_khs_cluster_wd_dim { \hfil #1 \hfil }
  }

\cs_new_protected:Npn \__font_khs_pair_row:nn #1#2
  {
    \bool_if:NF \l__font_khs_first_row_bool
      { \kern \l__font_khs_row_gap_dim }
    \bool_set_false:N \l__font_khs_first_row_bool
    \hbox to \l__font_khs_cluster_wd_dim
      {
        \hbox to \l__font_khs_cell_wd_dim { \hss #1 \hss }
        \kern \l__font_khs_col_gap_dim
        \hbox to \l__font_khs_cell_wd_dim { \hss #2 \hss }
      }
  }

\cs_new_protected:Npn \__font_khs_render_rows_from:n #1
  {
    \int_step_inline:nnnn { #1 } { 2 } { \seq_count:N \l__font_khs_char_seq }
      {
        \int_compare:nNnTF { ##1 } = { \seq_count:N \l__font_khs_char_seq }
          { \__font_khs_center_row:n { \seq_item:Nn \l__font_khs_char_seq { ##1 } } }
          {
            \__font_khs_pair_row:nn
              { \seq_item:Nn \l__font_khs_char_seq { ##1 } }
              { \seq_item:Nn \l__font_khs_char_seq { \int_eval:n { ##1 + 1 } } }
          }
      }
  }

\cs_new_protected:Npn \__font_khs_render_cluster_body:
  {
    \__font_khs_measure_cells:
    \__font_khs_compute_cluster_rows:
    \int_compare:nNnTF { \seq_count:N \l__font_khs_char_seq } = { 1 }
      { \hbox_set:Nn \l__font_khs_cluster_box { \seq_item:Nn \l__font_khs_char_seq { 1 } } }
      {
        \vbox_set:Nn \l__font_khs_cluster_box
          {
            \offinterlineskip
            \bool_set_true:N \l__font_khs_first_row_bool
            \bool_if:NTF \l__font_khs_type_b_bool
              {
                \__font_khs_center_row:n { \seq_item:Nn \l__font_khs_char_seq { 1 } }
                \int_compare:nNnT { \seq_count:N \l__font_khs_char_seq } > { 1 }
                  { \__font_khs_render_rows_from:n { 2 } }
              }
              { \__font_khs_render_rows_from:n { 1 } }
          }
      }
    \dim_set:Nn \l__font_khs_actual_ht_dim
      { \box_ht:N \l__font_khs_cluster_box + \box_dp:N \l__font_khs_cluster_box }
    \dim_set:Nn \l__font_khs_target_ht_dim
      {
        \int_max:nn { 1 } { \l__font_khs_run_max_rows_int }
        \l__font_khs_cell_ht_dim
        +
        \int_eval:n { \int_max:nn { 1 } { \l__font_khs_run_max_rows_int } - 1 }
        \l__font_khs_row_gap_dim
      }
    \dim_set:Nn \l__font_khs_raise_dim
      { \dim_eval:n { (\l__font_khs_target_ht_dim - \l__font_khs_actual_ht_dim) / 2 } }
    \raisebox{\dim_use:N \l__font_khs_raise_dim}{\box_use_drop:N \l__font_khs_cluster_box}
  }

\cs_new_protected:Npn \__font_khs_begin_cluster:
  {
    \seq_clear:N \l__font_khs_char_seq
    \bool_set_false:N \l__font_khs_type_b_bool
  }

\cs_new_protected:Npn \__font_khs_push_char:n #1
  {
    \seq_put_right:Nn \l__font_khs_char_seq { #1 }
  }

\cs_new_protected:Npn \__font_khs_mark_type_b:
  {
    \bool_set_true:N \l__font_khs_type_b_bool
  }

\cs_new_protected:Npn \__font_khs_space:
  { \hskip \l__font_khs_space_gap_dim }

\cs_new_protected:Npn \__font_khs_parse_cluster:n #1
  {
    \__font_khs_begin_cluster:
    \tl_map_inline:nn {#1}
      {
        \str_if_eq:eeTF
          { \tl_to_str:n {##1} }
          { \tl_to_str:V \c__font_khs_filler_tl }
          {
            \int_compare:nNnT { \seq_count:N \l__font_khs_char_seq } = { 1 }
              { \__font_khs_mark_type_b: }
          }
          { \__font_khs_push_char:n {##1} }
      }
  }

\cs_new_protected:Npn \__font_khs_compute_run_max_rows:
  {
    \int_set:Nn \l__font_khs_run_max_rows_int { 1 }
    \seq_map_inline:Nn \l__font_khs_cluster_seq
      {
        \tl_if_blank:nF {##1}
          {
            \__font_khs_parse_cluster:n {##1}
            \__font_khs_compute_cluster_rows:
            \int_set:Nn \l__font_khs_run_max_rows_int
              {
                \int_max:nn
                  { \l__font_khs_run_max_rows_int }
                  { \l__font_khs_cluster_rows_int }
              }
          }
      }
  }

\cs_new_protected:Npn \__font_khs_render_text:n #1
  {
    \regex_split:nnN { \s+ } {#1} \l__font_khs_cluster_seq
    \__font_khs_compute_run_max_rows:
    \bool_set_true:N \l__font_khs_first_cluster_bool
    \seq_map_inline:Nn \l__font_khs_cluster_seq
      {
        \tl_if_blank:nF {##1}
          {
            \bool_if:NF \l__font_khs_first_cluster_bool
              { \__font_khs_space: }
            \bool_set_false:N \l__font_khs_first_cluster_bool
            \__font_khs_parse_cluster:n {##1}
            \__font_khs_render_cluster_body:
          }
      }
  }

\cs_new_protected:Npn \__font_define_public_command_khitan:nn #1#2
  {
    \exp_args:Nc \RenewDocumentCommand {#1} { +m }
      {
        {
          \use:c { __font_prepare_local_#2: }
          \__font_khs_render_text:n { ##1 }
        }
      }

    \exp_args:Nc \RenewDocumentCommand { #1block } { +m }
      {
        {
          \use:c { __font_prepare_local_#2: }
          \noindent
          \__font_khs_render_text:n { ##1 }
        }
      }
  }

\cs_new_protected:Npn \__font_define_special_khitan_small:
  {
    \sys_if_engine_luatex:TF
      {
        \tl_clear:N \l__font_script_tl
        \tl_clear:N \l__font_language_tl
        \tl_set:Nn \l__font_backend_tl { generic }
        \__font_special_use_generic:
        \exp_args:NVV \__font_define_public_command_khitan:nn
          \l__font_command_tl
          \l__font_name_tl
      }
      {
        \PackageWarning{font-defaults}
          {Khitan~Small~Script~stacking~is~currently~supported~only~under~LuaLaTeX;~falling~back~to~linear~local~font~rendering~for~`\tl_use:N \l__font_name_tl`}
        \tl_clear:N \l__font_script_tl
        \tl_clear:N \l__font_language_tl
        \tl_set:Nn \l__font_backend_tl { generic }
        \__font_special_use_generic:
      }
  }

\ExplSyntaxOff
