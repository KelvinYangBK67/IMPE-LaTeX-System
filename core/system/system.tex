\cs_new_protected:Npn \__nextsystem_resolve_defaults:
  {
    \tl_set_eq:NN \l__nextsystem_effective_layout_tl \l__nextsystem_layout_tl
    \tl_set_eq:NN \l__nextsystem_effective_globalfonts_tl \l__nextsystem_globalfonts_tl

    \tl_if_blank:VT \l__nextsystem_effective_layout_tl
      {
        \cs_if_exist:NT \NextDefaultLayout
          { \tl_set:Nx \l__nextsystem_effective_layout_tl { \NextDefaultLayout } }
      }

    \tl_if_blank:VT \l__nextsystem_effective_globalfonts_tl
      {
        \cs_if_exist:NT \NextDefaultGlobalFonts
          { \tl_set:Nx \l__nextsystem_effective_globalfonts_tl { \NextDefaultGlobalFonts } }
      }
  }

\cs_new_protected:Npn \__nextsystem_apply_ui:
  {
    \bool_if:NF \g__nextsystem_ui_loaded_bool
      {
        \cs_if_exist:NTF \NextSystemUiTag
          { \tl_set:Nx \l__nextsystem_ui_tl { \NextSystemUiTag } }
          { \tl_set:Nn \l__nextsystem_ui_tl { en } }

        \str_if_eq:VnT \l__nextsystem_ui_tl { zh }
          {
            \bool_if:NTF \g__nextsystem_installed_bool
              { \input{core/system/ui_zh_internal.tex} }
              {
                \cs_if_exist:NTF \NextSystemRoot
                  { \expandafter\subimport\expandafter{\NextSystemRoot core/system/}{ui_zh_internal.tex} }
                  { \subimport{./}{ui_zh_internal.tex} }
              }
          }
        \bool_gset_true:N \g__nextsystem_ui_loaded_bool
      }
  }

\cs_new_protected:Npn \__nextsystem_apply_class_defaults:
  {
    \bool_if:NF \g__nextsystem_class_defaults_loaded_bool
      {
        \cs_if_exist:NT \NextDefaultLayout
          {
            \tl_set:Nx \l_tmpa_tl { \NextDefaultLayout }
            \tl_if_blank:VF \l_tmpa_tl
              { \exp_args:NV \UseLayout \l_tmpa_tl }
          }

        \cs_if_exist:NT \NextDefaultGlobalFonts
          {
            \tl_set:Nx \l_tmpa_tl { \NextDefaultGlobalFonts }
            \tl_if_blank:VF \l_tmpa_tl
              { \exp_args:NV \UseFonts \l_tmpa_tl [global] }
          }

        \__nextsystem_apply_ui:
        \bool_gset_true:N \g__nextsystem_class_defaults_loaded_bool
      }
  }

\keys_define:nn { nextsystem/setup }
  {
    layout .tl_set:N = \l__nextsystem_layout_tl,
    globalfonts .tl_set:N = \l__nextsystem_globalfonts_tl,
    mainfonts .meta:n = { globalfonts = {#1} },
    fonts .tl_set:N = \l__nextsystem_fonts_tl,
    features .tl_set:N = \l__nextsystem_features_tl,
  }

\cs_new_protected:Npn \__nextsystem_setup_apply:
  {
    \__nextsystem_resolve_defaults:

    \tl_if_blank:VF \l__nextsystem_effective_layout_tl
      { \exp_args:NV \UseLayout \l__nextsystem_effective_layout_tl }

    \tl_if_blank:VF \l__nextsystem_effective_globalfonts_tl
      { \exp_args:NV \UseFonts \l__nextsystem_effective_globalfonts_tl [global] }

    \tl_if_blank:VF \l__nextsystem_fonts_tl
      { \exp_args:NV \UseFonts \l__nextsystem_fonts_tl }

    \__nextsystem_apply_ui:

    \tl_if_blank:VF \l__nextsystem_features_tl
      { \exp_args:NV \UseFeatures \l__nextsystem_features_tl }
  }

\NewDocumentCommand{\UseTemplateSet}{m}
  {
    \tl_clear:N \l__nextsystem_layout_tl
    \tl_clear:N \l__nextsystem_globalfonts_tl
    \tl_clear:N \l__nextsystem_fonts_tl
    \tl_clear:N \l__nextsystem_features_tl
    \keys_set:nn { nextsystem/setup } {#1}
    \__nextsystem_setup_apply:
  }
