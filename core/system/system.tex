% Unified public entry for the next-generation template system.
% This file wires the stable subsystem entries together.

\RequirePackage{expl3}
\RequirePackage{xparse}
\RequirePackage{import}

\makeatletter
\@ifundefined{ifNextSystemInstalled}
  {\newif\ifNextSystemInstalled\NextSystemInstalledfalse}
  {}
\makeatother

\ifNextSystemInstalled
  \input{core/layout/system.tex}
  \input{core/fonts/system.tex}
  \input{core/features/system.tex}
\else
  \subimport{../layout/}{system.tex}
  \subimport{../fonts/}{system.tex}
  \subimport{../features/}{system.tex}
\fi

\ExplSyntaxOn

\bool_new:N \g__nextsystem_installed_bool
\cs_if_exist:NTF \ifNextSystemInstalled
  {
    \ifNextSystemInstalled
      \bool_gset_true:N \g__nextsystem_installed_bool
    \else
      \bool_gset_false:N \g__nextsystem_installed_bool
    \fi
  }
  { \bool_gset_false:N \g__nextsystem_installed_bool }

\tl_new:N \l__nextsystem_layout_tl
\tl_new:N \l__nextsystem_globalfonts_tl
\tl_new:N \l__nextsystem_fonts_tl
\tl_new:N \l__nextsystem_features_tl
\tl_new:N \l__nextsystem_ui_tl

\keys_define:nn { nextsystem/setup }
  {
    layout .tl_set:N = \l__nextsystem_layout_tl,
    globalfonts .tl_set:N = \l__nextsystem_globalfonts_tl,
    mainfonts .meta:n = { globalfonts = {#1} },
    fonts .tl_set:N = \l__nextsystem_fonts_tl,
    features .tl_set:N = \l__nextsystem_features_tl,
  }

\cs_new_protected:Npn \__nextsystem_setup_apply:
  {
    \tl_if_blank:VF \l__nextsystem_layout_tl
      { \exp_args:NV \UseLayout \l__nextsystem_layout_tl }

    \tl_if_blank:VF \l__nextsystem_globalfonts_tl
      { \exp_args:NV \UseFonts \l__nextsystem_globalfonts_tl [global] }

    \tl_if_blank:VF \l__nextsystem_fonts_tl
      { \exp_args:NV \UseFonts \l__nextsystem_fonts_tl }

    \cs_if_exist:NTF \NextSystemUiTag
      { \tl_set:Nx \l__nextsystem_ui_tl { \NextSystemUiTag } }
      { \tl_set:Nn \l__nextsystem_ui_tl { en } }

    \str_if_eq:VnT \l__nextsystem_ui_tl { zh }
      {
        \bool_if:NTF \g__nextsystem_installed_bool
          { \input{core/system/ui_zh_internal.tex} }
          {
            \cs_if_exist:NTF \NextSystemRoot
              { \expandafter\subimport\expandafter{\NextSystemRoot core/system/}{ui_zh_internal.tex} }
              { \subimport{./}{ui_zh_internal.tex} }
          }
      }

    \tl_if_blank:VF \l__nextsystem_features_tl
      { \exp_args:NV \UseFeatures \l__nextsystem_features_tl }
  }

\NewDocumentCommand{\UseTemplateSet}{m}
  {
    \tl_clear:N \l__nextsystem_layout_tl
    \tl_clear:N \l__nextsystem_globalfonts_tl
    \tl_clear:N \l__nextsystem_fonts_tl
    \tl_clear:N \l__nextsystem_features_tl
    \keys_set:nn { nextsystem/setup } {#1}
    \__nextsystem_setup_apply:
  }

\NewDocumentCommand{\UseNextSystem}{m}
  { \UseTemplateSet{#1} }

\ExplSyntaxOff
