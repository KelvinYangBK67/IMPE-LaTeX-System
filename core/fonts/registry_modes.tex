    \__font_family_reset:
    \keys_set:nn { font/family } {#1}

    \tl_if_blank:VTF \l__font_family_id_tl
      {
        \PackageError{font-family-registry}
          {Missing~id~field}
          {Set~id~=...~when~registering~a~font~family.}
      }
      { }

    \tl_if_blank:VTF \l__font_family_default_mode_tl
      { \tl_set:Nn \l__font_family_default_mode_tl { local } }
      { }

    \__font_family_validate_mode:n { \tl_use:N \l__font_family_default_mode_tl }

    \prop_put:NVV \g__font_family_default_prop \l__font_family_id_tl \l__font_family_default_mode_tl

    \tl_if_blank:VTF \l__font_family_local_tl
      {
        \tl_if_blank:VF \l__font_family_flat_decl_tl
          { \tl_set:NV \l__font_family_local_tl \l__font_family_flat_decl_tl }
      }
      { }

    \tl_if_blank:VTF \l__font_family_range_profile_tl
      { }
      { \prop_put:NVV \g__font_family_range_profile_prop \l__font_family_id_tl \l__font_family_range_profile_tl }

    \tl_if_blank:VTF \l__font_family_global_route_tl
      { \tl_set:Nn \l__font_family_global_route_tl { fontspec-range } }
      { }
    \prop_put:NVV \g__font_family_global_route_prop \l__font_family_id_tl \l__font_family_global_route_tl

    \tl_if_blank:VTF \l__font_family_local_tl
      { }
      { \prop_put:NVV \g__font_family_local_prop \l__font_family_id_tl \l__font_family_local_tl }

    \tl_if_blank:VTF \l__font_family_global_tl
      { }
      { \prop_put:NVV \g__font_family_global_prop \l__font_family_id_tl \l__font_family_global_tl }
  }

  }

\cs_new_protected:Npn \__font_family_use_auto_global:n #1
  {
    \prop_get:NnNTF \g__font_family_global_route_prop {#1} \l_tmpa_tl
      { }
      { \tl_set:Nn \l_tmpa_tl { fontspec-range } }
    \str_if_eq:VnTF \l_tmpa_tl { none }
      {
        \tl_clear:N \l__font_family_owned_range_keys_tl
        \tl_clear:N \l__font_family_owned_unicode_blocks_tl
      }
      {
        \__font_family_claim_profile:n {#1}
        \tl_if_blank:VF \l__font_family_owned_unicode_blocks_tl
          {
            \prop_if_in:NnF \g__font_family_loaded_prop { #1 / auto-global }
              {
                \prop_get:NnNTF \g__font_family_global_prop {#1} \l_tmpa_tl
                  {
                    \tl_set:Nx \l_tmpb_tl
                      {
                        scope = global, familyid = #1,
                        \exp_not:V \l_tmpa_tl
                        , unicodeblocks = { \exp_not:V \l__font_family_owned_unicode_blocks_tl }
                      }
                    \exp_args:NV \FontDeclare \l_tmpb_tl
                    \prop_put:Nnn \g__font_family_loaded_prop { #1 / auto-global } { loaded }
                  }
                  {
                    \prop_get:NnNT \g__font_family_local_prop {#1} \l_tmpa_tl
                      {
                        \tl_set:Nx \l_tmpb_tl
                          {
                            scope = global, familyid = #1,
                            \exp_not:V \l_tmpa_tl
                            , name = { #1 _global }
                            , unicodeblocks = { \exp_not:V \l__font_family_owned_unicode_blocks_tl }
                          }
                        \exp_args:NV \FontDeclare \l_tmpb_tl
                        \prop_put:Nnn \g__font_family_loaded_prop { #1 / auto-global } { loaded }
                      }
                  }
              }
          }
      }
  }

\cs_new_protected:Npn \__font_family_use_auto:n #1
  {
    \prop_if_in:NnTF \g__font_family_default_prop {#1}
      {
        \prop_if_in:NnT \g__font_family_local_prop {#1}
          {
            \prop_get:NnN \g__font_family_local_prop {#1} \l_tmpa_tl
            \__font_family_use_declaration:nnV {#1} { local } \l_tmpa_tl
          }
        \prop_get:NnN \g__font_family_default_prop {#1} \l_tmpa_tl
        \str_if_eq:VnT \l_tmpa_tl { global }
          {
            \prop_if_in:NnT \g__font_family_global_prop {#1}
              {
                \prop_get:NnN \g__font_family_global_prop {#1} \l_tmpb_tl
                \__font_family_use_declaration:nnV {#1} { global } \l_tmpb_tl
              }
          }
        \__font_family_use_auto_global:n {#1}
        \tl_if_blank:VT \l__font_family_owned_unicode_blocks_tl
          {
            \str_if_eq:VnF \l_tmpa_tl { global }
              {
                \prop_if_in:NnT \g__font_family_global_prop {#1}
                  {
                    \prop_get:NnN \g__font_family_global_prop {#1} \l_tmpb_tl
                    \__font_family_use_declaration:nnV {#1} { global } \l_tmpb_tl
                  }
              }
          }
      }
      {
        \PackageError{font-family-registry}
          {Unknown~font~family~id}
          {Register~family~`#1'~with~\string\FontRegisterFamily\ first.}
      }
  }

\cs_new_protected:Npn \__font_family_use_loader:nn #1#2
  {
    \str_if_eq:nnTF {#2} { local }
      {
        \prop_if_in:NnTF \g__font_family_local_prop {#1}
          {
            \prop_get:NnN \g__font_family_local_prop {#1} \l_tmpa_tl
            \__font_family_use_declaration:nnV {#1} { local } \l_tmpa_tl
          }
          {
            \PackageError{font-family-registry}
              {Unknown~font~family~mode}
              {Family~`#1'~does~not~provide~mode~`local'.}
          }
      }
      {
        \str_if_eq:nnTF {#2} { global }
          {
            \prop_if_in:NnTF \g__font_family_global_prop {#1}
              {
                \prop_get:NnN \g__font_family_global_prop {#1} \l_tmpa_tl
                \__font_family_use_declaration:nnV {#1} { global } \l_tmpa_tl
              }
              {
                \prop_if_in:NnTF \g__font_family_default_prop {#1}
                  {
                    \__font_family_use_auto_global:n {#1}
                    \tl_if_blank:VT \l__font_family_owned_unicode_blocks_tl
                      {
                        \PackageError{font-family-registry}
                          {Font~family~has~no~global~mode}
                          {Family~`#1'~does~not~provide~a~global~block~or~rangeprofile.}
                      }
                  }
                  {
                    \PackageError{font-family-registry}
                      {Unknown~font~family~id}
                      {Register~family~`#1'~with~\string\FontRegisterFamily\ first.}
                  }
              }
          }
          {
            \PackageError{font-family-registry}
              {Unknown~font~family~mode}
              {Use~local~or~global.}
          }
      }
  }
