\cs_new_protected:Npn \__font_import_special_module:
  {
    \exp_args:NV \__font_special_is_builtin:nTF \l__font_special_module_tl
      { }
      {
        \seq_if_in:NVF \g__font_loaded_special_modules_seq \l__font_special_module_tl
          {
            \bool_if:NTF \l__font_nextsystem_installed_bool
              {
                \use:x
                  {
                    \exp_not:N \subimport
                      { modules/fonts/ }
                      { \l__font_special_module_tl .tex }
                  }
              }
              {
                \use:x
                  {
                    \exp_not:N \subimport
                      { ../../modules/fonts/ }
                      { \l__font_special_module_tl .tex }
                  }
              }
            \seq_gput_right:NV \g__font_loaded_special_modules_seq \l__font_special_module_tl
          }
      }
  }
