\cs_new_protected:Npn \__font_define_special_local:
  {
    \seq_if_in:NVF \g__font_loaded_special_modules_seq \l__font_special_module_tl
      {
        \use:x
          {
            \exp_not:N \subimport
              { ../../modules/fonts/ }
              { \l__font_special_module_tl .tex }
          }
        \seq_gput_right:NV \g__font_loaded_special_modules_seq \l__font_special_module_tl
      }
    \tl_set:Nx \l__font_special_entry_tl
      { __font_define_special_\l__font_special_module_tl : }
    \exp_args:NV \cs_if_exist:cTF \l__font_special_entry_tl
      { \exp_args:NV \use:c \l__font_special_entry_tl }
      {
        \PackageError{font-defaults}
          {Special~module~`\tl_use:N \l__font_special_module_tl`~did~not~define~its~entry~point}
          {Define~\string\__font_define_special_\tl_use:N \l__font_special_module_tl :~inside~that~module.}
      }
  }
