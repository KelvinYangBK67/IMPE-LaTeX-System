\usepackage{fancyhdr}

\pagestyle{fancy}
\fancyhf{}

% 頁碼外側
\fancyfoot[LE,RO]{\thepage}

\fancypagestyle{plain}{
  \fancyhf{}
  \fancyhead[LE]{\nouppercase{\leftmark}}
  \fancyhead[RO]{\nouppercase{\rightmark}}
  \fancyfoot[LE,RO]{\thepage}
}
