\NewDocumentCommand{\FontRegister}{m}
  {
    \__font_registry_reset:
    \keys_set:nn { font/register } {#1}
    \tl_if_blank:VTF \l__font_registry_id_tl
      { \PackageError{font-registry}{Missing~id~field}{Set~id~=...~when~registering~a~font.} }
      { }
    \__font_registry_build_decl:
    \prop_put:NVV \g__font_registry_decl_prop \l__font_registry_id_tl \l__font_registry_decl_tl
  }

\cs_new_protected:Npn \__font_registry_use:n #1
  {
    \prop_if_in:NnTF \g__font_registry_decl_prop {#1}
      {
        \prop_if_in:NnTF \g__font_registry_loaded_prop {#1}
          { }
          {
            \prop_get:NnN \g__font_registry_decl_prop {#1} \l_tmpa_tl
            \exp_args:NV \FontDeclare \l_tmpa_tl
            \prop_put:Nnn \g__font_registry_loaded_prop {#1} { loaded }
          }
      }
      {
        \PackageError{font-registry}{Unknown~font~id~`#1`}{Register~the~font~with~\string\FontRegister\ before~using~it.}
      }
  }

\NewDocumentCommand{\UseFont}{m}
  { \__font_registry_use:n {#1} }

\NewDocumentCommand{\UseFonts}{m}
  { \clist_map_inline:nn {#1} { \__font_registry_use:n {##1} } }
