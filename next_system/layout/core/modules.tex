\cs_new_protected:Npn \__layout_apply_component:n #1
  {
    \prop_if_in:NnTF \g__layout_component_loaded_prop {#1}
      { }
      {
        \cs_if_exist:cTF { __layout_component_#1: }
          {
            \use:c { __layout_component_#1: }
            \prop_put:Nnn \g__layout_component_loaded_prop {#1} { loaded }
          }
          {
            \PackageError{layout-modules}
              {Unknown~internal~layout~component~`#1`}
              {Define~the~component~before~using~it~from~a~preset.}
          }
      }
  }

\cs_new_protected:Npn \__layout_apply_component_list:n #1
  {
    \tl_if_blank:nTF {#1}
      { }
      { \clist_map_inline:nn {#1} { \__layout_apply_component:n {##1} } }
