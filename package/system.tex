% Public repository entry for the next-generation template system.
% This wrapper anchors repository-local paths and then delegates to core.

\RequirePackage{expl3}
\RequirePackage{xparse}
\RequirePackage{import}

\makeatletter
\providecommand{\NextSystemRoot}{}
\providecommand{\CurrentFilePath}{}
\ifx\CurrentFilePath\@empty
  \ifx\NextSystemPackageRoot\@undefined
    \ifx\import@path\@undefined
      \xdef\NextSystemRoot{./}
    \else
      \xdef\NextSystemRoot{\import@path}
    \fi
  \else
    \ifx\NextSystemPackageRoot\@empty
      \ifx\import@path\@undefined
        \xdef\NextSystemRoot{./}
      \else
        \xdef\NextSystemRoot{\import@path}
      \fi
    \else
      \xdef\NextSystemRoot{\NextSystemPackageRoot}
    \fi
  \fi
\else
  \xdef\NextSystemRoot{\CurrentFilePath}
\fi
\makeatother

\providecommand{\CatalogFontRoot}{}
\NewDocumentCommand{\SetCatalogFontRoot}{m}
  {\def\CatalogFontRoot{#1}}
\InputIfFileExists{nextsystem.local.tex}{}{}

\ExplSyntaxOn
\tl_if_blank:eT { \CatalogFontRoot }
  {
    \tl_set:Nx \l_tmpa_tl { \NextSystemRoot assets/fonts }
    \exp_args:NV \SetCatalogFontRoot \l_tmpa_tl
  }
\ExplSyntaxOff

\expandafter\subimport\expandafter{\NextSystemRoot core/system/}{system.tex}
