% Public repository entry for the next-generation template system.
% This wrapper anchors repository-local paths and then delegates to core.

\RequirePackage{expl3}
\RequirePackage{xparse}
\RequirePackage{import}
\RequirePackage{xstring}

\makeatletter
\@ifundefined{ifNextSystemInstalled}
  {\newif\ifNextSystemInstalled\NextSystemInstalledfalse}
  {}
\providecommand{\NextSystemRoot}{}
\providecommand{\CurrentFilePath}{}
\ifNextSystemInstalled
  \xdef\NextSystemRoot{}
\else
  \ifx\CurrentFilePath\@empty
    \ifx\import@path\@undefined
      \xdef\NextSystemRoot{../}
    \else
      \StrBefore{\import@path}{package/}[\NextSystemRoot]
      \ifx\NextSystemRoot\@empty
        \edef\NextSystemRoot{\import@path}
      \fi
    \fi
  \else
    \StrBefore{\CurrentFilePath}{package/}[\NextSystemRoot]
    \ifx\NextSystemRoot\@empty
      \edef\NextSystemRoot{\CurrentFilePath}
      \IfEndWith{\NextSystemRoot}{/}{}{
        \edef\NextSystemRoot{\NextSystemRoot/}
      }
    \fi
  \fi
\fi
\makeatother

\providecommand{\CatalogFontRoot}{}
\NewDocumentCommand{\SetCatalogFontRoot}{m}
  {\def\CatalogFontRoot{#1}}
\ifNextSystemInstalled
  \InputIfFileExists{nextsystem.local.tex}{}{}
\fi

\makeatletter
\ifx\CatalogFontRoot\@empty
  \ifNextSystemInstalled
    \edef\CatalogFontRoot{assets/fonts}
  \else
    \ifx\import@path\@undefined
      \edef\CatalogFontRoot{../assets/fonts}
    \else
      \StrBefore{\import@path}{package/}[\CatalogFontRoot]
      \ifx\CatalogFontRoot\@empty
        \edef\CatalogFontRoot{\import@path}
      \fi
      \IfEndWith{\CatalogFontRoot}{/}{}{
        \edef\CatalogFontRoot{\CatalogFontRoot/}
      }
      \edef\CatalogFontRoot{\CatalogFontRoot assets/fonts}
      \StrSubstitute{\CatalogFontRoot}{ }{}[\CatalogFontRoot]
    \fi
  \fi
\fi
\makeatother

\ifNextSystemInstalled
  \input{core/system/system.tex}
\else
  \subimport{../core/system/}{system.tex}
\fi
